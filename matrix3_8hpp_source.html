188 Matrix3<T> Matrix3<T>::inverse() const {
189 T det = m[0][0] * (m[1][1] * m[2][2] - m[1][2] * m[2][1]) -
190 m[0][1] * (m[1][0] * m[2][2] - m[1][2] * m[2][0]) +
191 m[0][2] * (m[1][0] * m[2][1] - m[1][1] * m[2][0]);
192
193 if (det == 0) {
194 invokeError<LogicError>("Matrix is singular and cannot be inverted.");
195 }
196
197 Matrix3<T> result;
198 result.m[0][0] = (m[1][1] * m[2][2] - m[1][2] * m[2][1]) / det;
199 result.m[0][1] = (m[0][2] * m[2][1] - m[0][1] * m[2][2]) / det;
200 result.m[0][2] = (m[0][1] * m[1][2] - m[0][2] * m[1][1]) / det;
201
202 result.m[1][0] = (m[1][2] * m[2][0] - m[1][0] * m[2][2]) / det;
203 result.m[1][1] = (m[0][0] * m[2][2] - m[0][2] * m[2][0]) / det;
204 result.m[1][2] = (m[0][2] * m[1][0] - m[0][0] * m[1][2]) / det;
