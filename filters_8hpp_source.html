323 template<typename InputType, typename OutputType, typename FilterData>
324 void Filter<InputType, OutputType, FilterData>::applyConcurrent(FilterableBuffer<InputType> *inputBuffer, FilterableBuffer<OutputType> *outputBuffer) {
325 if (!m_multiFilterFunction) return;
326
327 const u32 size = inputBuffer->getSize();
328 const u32 numThreads = std::thread::hardware_concurrency();
329 const u32 elementsPerThread = (size + numThreads - 1) / numThreads;
330
331 std::vector<std::future<void>> futures;
332 futures.reserve(numThreads);
333
334 for (u32 threadId = 0; threadId < numThreads; ++threadId) {
335 const u32 start = threadId * elementsPerThread;
336 const u32 end = std::min(start + elementsPerThread, size);
337
338 if (start < end) {
339 futures.push_back(std::async(std::launch::async, [&, start, end]() {
340 for (u32 i = start; i < end; ++i) {
341 m_multiFilterFunction(inputBuffer->operator[](i), outputBuffer->operator[](i), data);
342 }
343 }));
344 }
345 }
346
347 for (auto& future : futures) {
348 future.wait();
349 }
