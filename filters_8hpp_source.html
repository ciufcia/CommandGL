330 template<typename InputType, typename OutputType, typename FilterData>
331 void Filter<InputType, OutputType, FilterData>::applyConcurrent(FilterableBuffer<InputType> *inputBuffer, FilterableBuffer<OutputType> *outputBuffer) {
332 if (!m_multiFilterFunction) return;
333
334 const u32 size = inputBuffer->getSize();
335 const u32 numThreads = std::thread::hardware_concurrency();
336 const u32 elementsPerThread = (size + numThreads - 1) / numThreads;
337
338 std::vector<std::future<void>> futures;
339 futures.reserve(numThreads);
340
341 for (u32 threadId = 0; threadId < numThreads; ++threadId) {
342 const u32 start = threadId * elementsPerThread;
343 const u32 end = std::min(start + elementsPerThread, size);
344
345 if (start < end) {
346 futures.push_back(std::async(std::launch::async, [&, start, end]() {
347 for (u32 i = start; i < end; ++i) {
348 m_multiFilterFunction(inputBuffer->operator[](i), outputBuffer->operator[](i), data);
349 }
350 }));
351 }
352 }
353
354 for (auto& future : futures) {
355 future.wait();
356 }
