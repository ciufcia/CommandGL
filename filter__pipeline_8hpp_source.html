550 template<typename InputType, typename OutputType>
551 void FilterPipeline<InputType, OutputType>::run(FilterableBuffer<InputType> *inputBuffer, FilterableBuffer<OutputType> *outputBuffer, const filters::BaseData &baseData) {
552 if (!built) {
553 invokeError<LogicError>("Pipeline is not built");
554 }
555
556 if (m_filters.size() == 0) {
557 return;
558 }
559
560 for (const auto &filter : m_filters) {
561 filter->setBaseData(baseData);
562 filter->beforePipelineRun();
563 }
564
565 if (m_filters.size() == 1) {
566 m_filters[0]->apply(inputBuffer, outputBuffer);
567 return;
568 }
569
570 BaseFilterableBuffer *currentInput = inputBuffer;
571 BaseFilterableBuffer *currentOutput;
572
573 for (int i = 0; i < m_filters.size() - 1; ++i) {
574 currentOutput = m_buffers[i].buffer;
575
576 if (!currentOutput) {
577 if (m_buffers[i].type == std::type_index(typeid(InputType))) {
578 currentOutput = inputBuffer;
579 } else if (m_buffers[i].type == std::type_index(typeid(OutputType))) {
580 currentOutput = outputBuffer;
581 } else {
582 invokeError<LogicError>("Could not find a matching buffer");
583 }
584 }
585
586 m_filters[i]->apply(currentInput, currentOutput);
587 currentInput = currentOutput;
588 }
589
590 m_filters[m_filters.size() - 1]->apply(currentInput, outputBuffer);
591
592 for (const auto &filter : m_filters) {
593 filter->afterPipelineRun();
594 }
595 }
