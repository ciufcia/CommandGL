198 void FilterPipeline<InputType, OutputType>::run(FilterableBuffer<InputType> *inputBuffer, FilterableBuffer<OutputType> *outputBuffer, const filters::BaseData &baseData) {
199 if (!built) {
200 invokeError<LogicError>("Pipeline is not built");
201 }
202
203 if (m_filters.size() == 0) {
204 return;
205 }
206
207 for (const auto &filter : m_filters) {
208 filter->setBaseData(baseData);
209 filter->beforePipelineRun();
210 }
211
212 if (m_filters.size() == 1) {
213 m_filters[0]->apply(inputBuffer, outputBuffer);
214 return;
215 }
216
217 BaseFilterableBuffer *currentInput = inputBuffer;
218 BaseFilterableBuffer *currentOutput;
219
220 for (int i = 0; i < m_filters.size() - 1; ++i) {
221 currentOutput = m_buffers[i].buffer;
222
223 if (!currentOutput) {
224 if (m_buffers[i].type == std::type_index(typeid(InputType))) {
225 currentOutput = inputBuffer;
226 } else if (m_buffers[i].type == std::type_index(typeid(OutputType))) {
227 currentOutput = outputBuffer;
228 } else {
229 invokeError<LogicError>("Could not find a matching buffer");
230 }
231 }
232
233 m_filters[i]->apply(currentInput, currentOutput);
234 currentInput = currentOutput;
235 }
236
237 m_filters[m_filters.size() - 1]->apply(currentInput, outputBuffer);
238
239 for (const auto &filter : m_filters) {
240 filter->afterPipelineRun();
241 }
